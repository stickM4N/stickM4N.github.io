147 double128_t wstr_cast_to_double128(const_wcstring string, size_t length) {
148
149 size_t char_position = 0ul;
150 bool_t is_negative = string[char_position] == '-';
151 if (is_negative)
152 char_position++;
153
154 double128_t integer_part = 0.0l;
155 if (string[char_position] == L'0')
156 char_position++;
157 else if (wchar_is_numeric(string[char_position]))
158 do
159 (integer_part *= 10) += string[char_position++] - L'0';
160 while (wchar_is_numeric(string[char_position]));
161 else
162 throw cast_error("Given string is not a number. No integer part.");
163
164 double128_t decimal_part = 0.0l;
165 if (string[char_position] == L'.') {
166 char_position++;
167
168 if (not wchar_is_numeric(string[char_position]))
169 throw cast_error(
170 "Given string is not a number. Expected decimal values.");
171
172 double128_t factor = 0.1l;
173 do {
174 decimal_part += (string[char_position++] - L'0') * factor;
175 factor *= 0.1l;
176 } while (wchar_is_numeric(string[char_position]));
177 }
178
179 uint8_t exponent = 0;
180 bool_t negative_exponent = false;
181 if (string[char_position] == L'E' or string[char_position] == L'e') {
182 char_position++;
183
184 if (string[char_position] == L'+')
185 char_position++;
186 else if (string[char_position] == L'-') {
187 negative_exponent = true;
188 char_position++;
189 }
190
191 if (string[char_position] == L'0')
192 char_position++;
193 else if (wchar_is_numeric(string[char_position]))
194 do
195 exponent = exponent * 10 + (string[char_position++] - L'0');
196 while (wchar_is_numeric(string[char_position]));
197 else
198 throw cast_error(
199 "Given string is not a number. Invalid exponent.");
200 }
201
202 if (char_position != length)
203 throw cast_error("Given string is not a number. "
204 "Could not be parsed completely.");
205
206 double128_t number = integer_part + decimal_part;
207 if (negative_exponent)
208 for (uint8_t i = 0; i < exponent; i++)
209 number /= 10;
210 else // positive exponent
211 for (uint8_t i = 0; i < exponent; i++)
212 number *= 10;
213 if (is_negative)
214 number = -number;
215
216 return number;
217 }
218 double128_t wstr_cast_to_double128(const_wcstring string) {
219 return wstr_cast_to_double128(string, wstr_length(string));
220 }
