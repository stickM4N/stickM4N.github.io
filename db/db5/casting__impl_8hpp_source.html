41 template<>
42 const_cstring cast_to_cstr(int8_t value, cstring string) {
43 bool_t is_negative = value < 0;
44 if (is_negative)
45 value = -value;
46
47 size_t length = 1ul;
48 for (uint8_t i = 1u; value / i >= 10u; i *= 10u)
49 length++;
50
51 if (string == nullptr)
52 string = new_str(length + (is_negative ? 2u : 1u));
53
54 if (is_negative) {
55 string[0] = '-';
56 length++;
57 }
58 string[length--] = str_end;
59
60 for (; value > 0u; value /= 10u)
61 string[length--] = '0' + static_cast<char_t>(value % 10u);
62
63 return string;
64 }

215 template<>
216 const_cstring cast_to_cstr(float32_t value, cstring string) {
217 auto integer_part = static_cast<int64_t>(value);
218 auto decimal_part = (value - integer_part) * 10;
219
220 const_cstring integer_str = cast_to_cstr(integer_part, string);
221 size_t length = str_length(integer_str);
222
223 if (string == nullptr)
224 string = new_str(length + 10);
225
226 str_copy(integer_str, length, string, false);
227 delete_str(const_cast<cstring &>(integer_str));
228
229 if (decimal_part != 0.0) {
230 string[length++] = '.';
231
232 for (uint8_t i = 0u; i < 8 and decimal_part != 0.0l; i++) {
233 string[length++] = '0' + static_cast<char_t>(decimal_part);
234 (decimal_part -= static_cast<char_t>(decimal_part)) *= 10u;
235 }
236 }
237
238 string[length] = str_end;
239
240 return string;
241 }
